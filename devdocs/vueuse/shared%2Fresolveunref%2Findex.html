Get the value of value/ref/getter.

Usage 

import { resolveUnref } from '@vueuse/core'

const foo = ref('hi')

const a = resolveUnref(0) // 0
const b = resolveUnref(foo) // 'hi'
const c = resolveUnref(() => 'hi') // 'hi'

 * Get the value of value/ref/getter.
 */
